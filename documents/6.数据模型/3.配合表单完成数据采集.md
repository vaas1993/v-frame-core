# 配合表单完成数据采集
在大多数时候我们都是通过渲染表单来采集用户输入的信息，数据模型的数据源不单单可以通过调用接口赋值，还可以渲染一个表单然后通过表单获取用户输入的信息。

> 提示：表单的渲染需要开发者自己根据所使用的UI进行二次开发。

继承 AppModel 或其子类的模型类，在它的实例调用 getFormConfig 方法来获取一个用于渲染表单的配置：
```javascript
// 假定我们有用户模型
export default class User extends AuthUser {}

// 现在我们渲染一个修改密码表单，我们创建一个修改密码的表单类
export default class UserChangePasswordForm extends User {
    // 定义当前模型使用的 action 接口映射
    static ActionApi = 'user.password.update'
    
    // 预定义好模型的属性
    username
    password
    new_password
    re_password
    
    // 定义模型属性标签
    $fieldLabels = {
        username: '用户名',
        password: '原密码',
        new_password: '新密码',
        re_password: '确认密码'
    }
    
    // 定义表单校验规则
    $rules = {
        username: [
            {required: true}
        ],
        password: [
            {required: true}
        ],
        new_password: [
            {required: true, type: 'string', min: 6, max: 16}
        ],
        re_password: [
            {required: true, type: 'string', min: 6, max: 16},
            {
                validator: () => {
                    return this.new_password === this.re_password
                },
                message: '两次输入的密码不一致'
            }
        ],
    }

    /**
     * 定义表单渲染配置
     */
    getFormConfig() {
        return {
            username: {
                component: {
                    widget: 'input',
                },
                options: {
                    width: '100%'
                }
            },
            password: {
                component: {
                    widget: 'input',
                    options: {
                        type: 'password'
                    }
                },
                options: {
                    width: '100%'
                }
            },
            new_password: {
                component: {
                    widget: 'input',
                    options: {
                        type: 'password'
                    }
                },
                options: {
                    width: '100%'
                }
            },
            re_password: {
                component: {
                    widget: 'input',
                    options: {
                        type: 'password'
                    }
                },
                options: {
                    width: '100%'
                }
            },
        }
    }
}

// 在视图中，我们可以根据 getFormConfig 的返回值来渲染表单
// 具体实现可根据不同的UI实现

// 在视图的逻辑部分：
let formModel

// 使用全局用户模型的数据，实例化出修改密码的表单类
formModel = VFrame.getInstance().user.getInstanceTo(UserChangePasswordForm)

// 你也可以这么做：
formModel = new UserChangePasswordForm()
formModel.setSources(VFrame.getInstance().user.getSources())

// 在表单获取到数据后：

if( formModel.action() ) {
    alert('修改密码成功')
} else {
    // 输出错误信息
    console.log(formModel.$errors)
}
```
